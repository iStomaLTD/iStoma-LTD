CREATE PROCEDURE [dbo].[Recomandanti_V_GetListaCautare] 
	@tDenumire nVarChar(100)
AS
BEGIN

    SELECT nId, tdenumire FROM Recomandanti_V
	WHERE tDenumire LIKE @tDenumire collate Latin1_General_BIN2
	ORDER BY tDenumire asc

END
